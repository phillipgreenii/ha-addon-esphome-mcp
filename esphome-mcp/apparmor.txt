#include <tunables/global>

profile esphome-mcp flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Minimal capabilities for the s6-setuidgid drop-privs chain.
  capability chown,
  capability setuid,
  capability setgid,
  capability kill,

  # Executables we actually run
  /opt/** r,
  /opt/run.sh ix,
  /opt/server/** r,
  /usr/bin/python3* ix,
  /usr/bin/s6-setuidgid ix,
  /usr/local/bin/esphome ix,
  /usr/local/bin/platformio ix,
  # s6-overlay v3 wires its service-supervision chain via many helper
  # binaries (execlineb, s6-svscan, s6-svc, redirfd, fdmove, foreground,
  # importas, with-contenv, s6-overlay-suexec, etc.) that live under
  # /command/ and the /package/admin/s6-overlay/... symlink farm. The
  # init chain needs `mrix` (mmap + read + inherit + execute) — `ix`
  # alone causes downstream binaries to fail with `Permission denied`
  # because the ELF loader can't read/map their code segments. The
  # dangerous primitives (ptrace, mount, sys_admin) are explicitly
  # denied below.
  /command/** mrix,
  /package/admin/s6-overlay/command/** mrix,
  /package/admin/s6/command/** mrix,
  # Version-specific real targets. The /command/ and /package/admin/s6-overlay/
  # entries above are symlinks that ultimately resolve to /package/admin/s6-overlay-<version>/
  # and /package/admin/s6-<version>/. AppArmor evaluates the final target, so
  # these must be allowed explicitly.
  /package/admin/s6-overlay-*/** mrix,
  /package/admin/s6-*/** mrix,
  /usr/bin/with-contenv mrix,
  /usr/bin/bashio mrix,
  /usr/lib/bashio/** r,
  # /init needs `mr` in addition to `ix`. s6-overlay's stage-1 `/init`
  # is the container entrypoint; the kernel/shell needs to read (and
  # mmap) the file to dispatch the interpreter. With `ix` alone the
  # container fails to boot with
  # `/bin/sh: can't open '/init': Permission denied`. See also
  # `/etc/s6-overlay/** rix` below for the s6 supporting scripts.
  /init mrix,
  /etc/s6-overlay/** rix,
  # s6 runtime state — the AppArmor profile attaches to the WHOLE
  # container (Docker `--security-opt apparmor=esphome-mcp`), including
  # PID 1, which is s6-overlay's stage-1 `/init`. s6-svscan / s6-supervise
  # write supervisor state files (control fifos, event pipes) under these
  # paths, so they must be writable. The dangerous primitives (ptrace,
  # mount, sys_admin) are explicitly denied below.
  /run/s6/** rwk,
  /run/s6-rc/** rwk,
  /run/service/** rwk,

  # Python site-packages — required for ESPHome imports
  /usr/lib/python3.*/** r,
  /usr/local/lib/python3.*/** r,

  # Data + state
  /tmp/** rwk,
  /data/** rwk,
  # Note: defense against ESPHome subprocesses reading /data/auth_token is
  # provided at the Python layer by _scan_unsafe_includes (push_files
  # rejects any !include* targeting paths outside /share/esphome/). An
  # AppArmor `deny /data/auth_token rwx,` rule would also apply to run.sh
  # itself (which legitimately reads the token at boot), so we don't add
  # one here.
  /share/esphome/** rwk,

  # Network: outbound for ESPHome OTA + listening socket
  network inet stream,
  network inet6 stream,
  network inet dgram,

  # Deny dangerous ops explicitly
  deny ptrace,
  deny mount,
  deny umount,
  deny /proc/sysrq-trigger w,
  deny /sys/** w,
  deny capability sys_admin,
  deny capability sys_module,
  deny capability net_admin,
  deny capability net_raw,
  deny /proc/*/mem rw,
  deny /proc/kcore rw,
  deny /sys/kernel/security/** rwx,

  signal (send,receive) peer=esphome-mcp,
}
