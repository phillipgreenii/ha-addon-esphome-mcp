#include <tunables/global>

profile esphome-mcp flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Minimal capabilities for the s6-setuidgid drop-privs chain.
  capability chown,
  capability setuid,
  capability setgid,
  capability kill,

  # Executables we actually run
  /opt/** r,
  /opt/run.sh ix,
  /opt/server/** r,
  /usr/bin/python3* ix,
  /usr/bin/s6-setuidgid ix,
  /usr/local/bin/esphome ix,
  /usr/local/bin/platformio ix,
  /command/s6-setuidgid ix,
  /usr/bin/with-contenv ix,
  /usr/bin/bashio ix,
  /usr/lib/bashio/** r,
  /init ix,

  # Python site-packages — required for ESPHome imports
  /usr/lib/python3.*/** r,
  /usr/local/lib/python3.*/** r,

  # Data + state
  /tmp/** rwk,
  /data/** rwk,
  # The auth token file is owner-only (mode 0600 from run.sh). Deny
  # explicitly so that ESPHome subprocesses (compile/validate/etc.) cannot
  # exfiltrate it via crafted YAML, regardless of how containment is
  # bypassed at the Python layer.
  deny /data/auth_token rwx,
  /share/esphome/** rwk,

  # Network: outbound for ESPHome OTA + listening socket
  network inet stream,
  network inet6 stream,
  network inet dgram,

  # Deny dangerous ops explicitly
  deny ptrace,
  deny mount,
  deny umount,
  deny /proc/sysrq-trigger w,
  deny /sys/** w,
  deny capability sys_admin,
  deny capability sys_module,
  deny capability net_admin,
  deny capability net_raw,
  deny /proc/*/mem rw,
  deny /proc/kcore rw,
  deny /sys/kernel/security/** rwx,

  signal (send,receive) peer=esphome-mcp,
}
